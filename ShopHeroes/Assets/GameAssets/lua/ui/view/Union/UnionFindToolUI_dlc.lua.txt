
--UnionFindToolUI_dlc

local CS_type_ObjList = typeof(CS.ObjList)

local btn_refresh
local obj_loading
local objList_unionHot
local obj_nothing

function awake()
	
	btn_refresh = self:GetButton("btn_refresh")
	obj_loading = self:GetObjByName("obj_loading")
	objList_unionHot = self:GetObjByName("objList_unionHot"):GetComponent(CS_type_ObjList).objList
	obj_nothing = self:GetObjByName("obj_nothing")
	
	btn_refresh:ButtonClickTween(function ()
		
			for i = 0, objList_unionHot.Count - 1  do
				objList_unionHot[i]:SetActive(false)
			end
			
			
			obj_nothing:SetActive(false)
			obj_loading:SetActive(true)
			
			local req = MsgType.Request_Union_List:New()
			--req.searchName = ""
			req:Send()
			
	end)
	
end
